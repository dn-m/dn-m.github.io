Usage

List Processing

Destructure an Array into head and tail

let emptyArray = []
let (head, tail) = emptyArray.destructured // nil

let array = [1,2,3]
let (head, tail) = array.destructured // (1, [2,3])

Construct an Array from a head and tail

let head = 5
let tail = [10,15]
let list = head + tail  // [5,10,15]
